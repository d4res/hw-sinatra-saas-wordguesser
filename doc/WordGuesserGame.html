def self.get_random_word
  require 'uri'
  require 'net/http'
  uri = URI('http://randomword.saasbook.info/RandomWord')
  Net::HTTP.new('randomword.saasbook.info').start do |http|
    return http.post(uri, '').body
  end
end

def self.guess(s)
  s.each_char do |char|
    @guesses += char if @word.insclues? char
  end
end
